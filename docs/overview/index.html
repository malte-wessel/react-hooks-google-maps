#The gist

import { useMap, GoogleMap } from 'react-hooks-google-maps';
const duesseldorf = {
lat: 51.22172,
lng: 6.77616,
};
const App: FC = () => {
// Load the Google Maps API and get a Map instance
const map = useMap(GOOGLE_MAPS_KEY);
// Pass it to the `GoogleMap` component and configure your map
return <GoogleMap zoom={13} center={duesseldorf} map={map} />;
};
